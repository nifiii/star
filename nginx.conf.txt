# 定义自定义日志格式，专门记录 API 请求详情
# 格式: [时间] 客户IP "请求方法 路径" 状态码 - 查询参数
log_format api_tracking '[$time_local] $remote_addr "$request_method $uri" $status - Query: "$args" - UA: "$http_user_agent"';

server {
    listen 80;
    # 匹配 localhost 和 任意 IP 访问
    server_name localhost CHANGE.ME _;
    
    root /var/www/html;
    index index.html;

    # Gzip Optimization
    gzip on;
    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_min_length 256;
    gzip_types text/plain text/css application/json application/javascript text/xml application/xml application/xml+rss text/javascript;

    # 1. 核心 SPA 配置
    location / {
        try_files $uri $uri/ /index.html;
    }

    # 2. 鉴权配置
    location = /auth_config.json {
        add_header Cache-Control "no-store, no-cache, must-revalidate, max-age=0";
        expires off;
    }

    # [Deprecated but kept for fallback] 静态 JSON 文件
    location ~ ^/daily_.*\.json$ {
        add_header Cache-Control "public, max-age=3600";
        expires 1h;
    }

    # 3. [NEW] API Proxy for Node.js Server
    # 将前端 /api/ 请求转发到后台 Express 服务
    location /api/ {
        # 启用 API 访问日志记录
        access_log /var/log/nginx/api_access.log api_tracking;

        proxy_pass http://127.0.0.1:3000/api/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    # 4. 静态资源
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg)$ {
        expires 1y;
        add_header Cache-Control "public, no-transform";
    }

    # 5. Google Gemini AI Proxy
    location ^~ /google-ai/ {
        resolver 127.0.0.11 8.8.8.8 ipv6=off valid=30s;
        rewrite ^/google-ai/(.*) /$1 break;
        proxy_pass https://generativelanguage.googleapis.com;
        proxy_ssl_server_name on;
        proxy_ssl_name generativelanguage.googleapis.com;
        proxy_set_header Host generativelanguage.googleapis.com;
        proxy_http_version 1.1;
        proxy_set_header Connection "";
        proxy_buffering off;
        proxy_cache off;
        chunked_transfer_encoding on;
        proxy_read_timeout 300s;
    }

    error_page 500 502 503 504 /50x.html;
    location = /50x.html {
        root /usr/share/nginx/html;
    }
}
