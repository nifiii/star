server {
    listen 80;
    server_name localhost;
    
    # 对应 Dockerfile 中的 web root
    root /var/www/html;
    index index.html;

    # 开启 Gzip 压缩
    gzip on;
    gzip_types text/plain text/css application/json application/javascript text/xml application/xml application/xml+rss text/javascript;

    # 1. 核心 SPA 配置
    # 所有 404 请求都重定向回 index.html，交给 React Router 处理
    location / {
        try_files $uri $uri/ /index.html;
    }

    # 2. 关键配置：auth_config.json 不缓存
    # 这个文件由后台脚本动态更新，必须保证前端每次获取的都是最新的
    location = /auth_config.json {
        add_header Cache-Control "no-store, no-cache, must-revalidate, proxy-revalidate, max-age=0";
        expires off;
    }

    # 3. 静态资源缓存配置 (js, css, images)
    # Vite 生成的文件带有 hash 文件名，可以长时间缓存
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg)$ {
        expires 1y;
        add_header Cache-Control "public, no-transform";
    }

    error_page 500 502 503 504 /50x.html;
    location = /50x.html {
        root /usr/share/nginx/html;
    }
}
